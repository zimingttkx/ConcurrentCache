# 第六章：基础设施层（日志、配置、信号）

## 章节预告

前面几章我们学习了应用层、协议层、存储层、网络层。现在问题是：

**网络层、协议层、应用层都需要输出日志、读取配置、处理信号。这些公共功能应该放在哪里？**

这就是**基础设施层**要解决的问题。

## 6.1 日志系统

### 6.1.1 为什么需要日志？

```
没有日志的问题：

┌─────────────────────────────────────────────────────────────────────────────┐
│                                                                             │
│  程序崩溃了，但你不知道发生了什么                                          │
│                                                                             │
│  $ ./server                                                             │
│  Segmentation fault (core dumped)                                         │
│                                                                             │
│  你只能看到"崩溃了"，但不知道：                                          │
│  - 崩溃在哪里？                                                          │
│  - 崩溃前的操作是什么？                                                 │
│  - 是什么数据导致的崩溃？                                               │
│                                                                             │
└─────────────────────────────────────────────────────────────────────────────┘

有日志的好处：

┌─────────────────────────────────────────────────────────────────────────────┐
│                                                                             │
│  $ ./server                                                             │
│                                                                             │
│  [INFO] 2026-04-24 10:30:15.123 [NETWORK] [12345] Server started on port 6379   │
│  [INFO] 2026-04-24 10:30:16.456 [NETWORK] [12346] Client connected: fd=5, ip=127.0.0.1   │
│  [DEBUG] 2026-04-24 10:30:17.789 [CACHE] [12345] GET request: key=name             │
│  [ERROR] 2026-04-24 10:30:17.890 [PROTOCOL] [12345] RESP parse error: unexpected char    │
│                                                                             │
│  你可以清楚地看到程序运行的每一步，定位问题更容易                        │
│                                                                             │
└─────────────────────────────────────────────────────────────────────────────┘
```

### 6.1.5 Logger V2 实现

**文件：** `src/base/log.h`

```cpp
// ==================== LogLevel 扩展 ====================

enum class LogLevel {
    TRACE,  // 新增：最细粒度，如函数入口、变量值
    DEBUG,  // 调试信息
    INFO,   // 一般信息
    WARN,   // 警告信息
    ERROR,  // 错误信息
    FATAL   // 新增：致命错误，程序无法继续
};

// ==================== Sink 抽象 ====================

/**
 * @brief Sink 基类 - 日志输出目标抽象
 */
class Sink {
public:
    virtual ~Sink() = default;
    virtual void write(const std::string& message) = 0;
    virtual void flush() = 0;
};

/**
 * @brief 控制台 Sink
 */
class ConsoleSink : public Sink {
public:
    void write(const std::string& message) override;
    void flush() override;
};

/**
 * @brief 文件 Sink（支持日志轮转）
 */
class FileSink : public Sink {
public:
    FileSink(const std::string& filepath, size_t maxSize, int maxFiles);
    ~FileSink();

    void write(const std::string& message) override;
    void flush() override;

private:
    void rotate();     // 日志轮转
    void cleanup();    // 清理多余历史文件

    std::string filepath_;
    size_t maxSize_;
    int maxFiles_;
    std::ofstream file_;
    std::mutex mutex_;  // 保护文件操作
};

// ==================== Logger 类 ====================

class Logger : public ConfigObserver {
public:
    static Logger& instance();

    // 配置
    void setLevel(LogLevel level);
    void setFile(const std::string& path);
    void setRotation(size_t maxSize, int maxFiles);

    // 日志写入（带模块）
    void log(const char* module, LogLevel level, const char* fmt, ...);

    // 刷新队列（阻塞直到队列为空）
    void flush();

    // ConfigObserver 实现
    void onConfigChange(const std::string& key, const std::string& value) override;

private:
    Logger();
    ~Logger();

    std::string formatMessage(const char* module, LogLevel level, const std::string& msg);
    void writerThread();  // 后台写入线程

    LogLevel level_;
    size_t maxFileSize_;
    int maxFiles_;
    bool stop_;
    std::string defaultModule_;

    std::vector<std::shared_ptr<Sink>> sinks_;  // Sink 列表
    std::queue<std::string> queue_;              // 日志队列
    std::mutex mutex_;
    std::condition_variable cv_;
    std::thread writerThread_;
};
```

### 6.2.3 ConfigObserver 观察者模式

```cpp
/**
 * @brief 配置观察者接口
 *
 * 当配置发生变化时，被观察的对象会调用 onConfigChange 通知观察者
 */
class ConfigObserver {
public:
    virtual ~ConfigObserver() = default;

    /**
     * @brief 配置变更通知
     * @param key 配置项名称
     * @param value 配置项新值
     */
    virtual void onConfigChange(const std::string& key,
                               const std::string& value) = 0;
};

/**
 * @brief Config 类 - 配置管理（单例）
 */
class Config : public ConfigObserver {
public:
    static Config& instance();

    // 加载配置
    bool load(const std::string& filename);

    // 观察者管理
    void addObserver(const std::string& key, ConfigObserver* observer);
    void removeObserver(const std::string& key, ConfigObserver* observer);

    // 热加载
    void reload();

    // 获取值
    std::string getString(const std::string& key,
                         const std::string& default_val = "");
    int getInt(const std::string& key, int default_val = 0);
    bool getBool(const std::string& key, bool default_val = false);

    // ConfigObserver 实现
    void onConfigChange(const std::string& key,
                       const std::string& value) override;

private:
    Config();
    ~Config();

    void notifyObservers(const std::string& key, const std::string& value);

    std::unordered_map<std::string, std::string> values_;
    std::unordered_map<std::string,
        std::vector<ConfigObserver*>> observers_;
    std::mutex mutex_;
    std::string configFile_;
};
```

## 6.3 信号处理

### 6.3.1 什么是信号？

```
信号 = 软件中断，通知进程发生了某个事件

常见信号：

┌─────────────────────────────────────────────────────────────────────────────┐
│                                                                             │
│  SIGINT (Ctrl+C)    ─ 终止进程，正常退出                                │
│  SIGTERM             ─ 终止进程，请求优雅退出                            │
│  SIGSEGV             ─ 段错误，程序崩溃                                  │
│  SIGPIPE             ─ 向已关闭的 socket 写数据                         │
│  SIGHUP              ─ 终端关闭                                          │
│                                                                             │
└─────────────────────────────────────────────────────────────────────────────┘
```

### 6.3.2 为什么需要信号处理？

```
没有信号处理的问题：

┌─────────────────────────────────────────────────────────────────────────────┐
│                                                                             │
│  $ ./server                                                             │
│  Server started...                                                       │
│  ...                                                                      │
│  ^C  ← 用户按 Ctrl+C                                                    │
│                                                                             │
│  程序直接退出，可能：                                                     │
│  - 正在写的日志没有刷新                                                   │
│  - 正在处理的请求被中断                                                   │
│  - 客户端连接没有优雅关闭                                                 │
│                                                                             │
└─────────────────────────────────────────────────────────────────────────────┘

有信号处理的好处：

┌─────────────────────────────────────────────────────────────────────────────┐
│                                                                             │
│  $ ./server                                                             │
│  Server started...                                                       │
│  ...                                                                      │
│  ^C                                                                      │
│                                                                             │
│  捕获 SIGINT：                                                           │
│  ┌─────────────────────────────────────────────────────────────────────┐ │
│  │  1. 设置退出标志 quit_ = true                                      │ │
│  │  2. EventLoop 检测到 quit_，退出循环                                 │ │
│  │  3. 等待所有连接处理完成                                            │ │
│  │  4. 刷新日志文件                                                    │ │
│  │  5. 释放资源                                                        │ │
│  │  6. 优雅退出                                                        │ │
│  └─────────────────────────────────────────────────────────────────────┘ │
│                                                                             │
└─────────────────────────────────────────────────────────────────────────────┘
```

### 6.3.3 V2 Signal 增强计划

```
V2 将增强信号处理：
┌─────────────────────────────────────────────────────────────────────────────┐
│                                                                             │
│  1. SIGSEGV 处理：崩溃时打印堆栈信息                                     │
│     - 使用 libunwind 或 glibc 的 backtrace                                │
│     - 输出函数调用栈，帮助定位问题                                        │
│                                                                             │
│  2. SIGPIPE 忽略：避免向关闭连接写入崩溃                                   │
│     - signal(SIGPIPE, SIG_IGN)                                           │
│                                                                             │
└─────────────────────────────────────────────────────────────────────────────┘
```

### 6.3.4 SignalHandler 实现

```cpp
class SignalHandler {
public:
    // 注册信号处理函数
    static void setup();

private:
    static void handleSignal(int sig);
};
```

**常用信号处理：**

```cpp
void SignalHandler::setup() {
    // SIGINT: Ctrl+C
    signal(SIGINT, handleSignal);

    // SIGTERM: kill 命令默认信号
    signal(SIGTERM, handleSignal);

    // SIGPIPE: 忽略，防止写已关闭 socket 崩溃
    signal(SIGPIPE, SIG_IGN);

    // V2 计划：SIGSEGV: 捕获段错误，打印堆栈
    // signal(SIGSEGV, handleSignal);
}

void SignalHandler::handleSignal(int sig) {
    switch (sig) {
        case SIGINT:
        case SIGTERM:
            LOG_INFO("MAIN", "Received signal %d, shutting down...", sig);
            // 设置退出标志，EventLoop 会优雅退出
            EventLoop::getInstance().quit();
            break;

        case SIGSEGV:
            LOG_FATAL("MAIN", "Segmentation fault! Stack trace:");
            // V2: 打印堆栈信息
            // printStackTrace();
            abort();  // 崩溃退出
            break;

        default:
            LOG_WARN("MAIN", "Received unexpected signal %d", sig);
            break;
    }
}
```

## 6.4 main.cpp 主函数

### 6.4.1 完整主函数

```cpp
#include "src/network/event_loop.h"
#include "src/network/socket.h"
#include "src/network/connection.h"
#include "src/command/command_factory.h"
#include "src/base/log.h"
#include "src/base/config.h"
#include "src/base/signal.h"

int main() {
    // ===== 1. 初始化日志 =====
    // 从配置文件读取日志级别，实现热加载
    Logger::instance().setLevel(LogLevel::INFO);
    Logger::instance().setFile("./logs/concurrentcache.log");
    Logger::instance().setRotation(100 * 1024 * 1024, 5);  // 100MB，保留5个

    LOG_INFO(NETWORK, "=== ConcurrentCache Server Starting ===");

    // ===== 2. 加载配置 =====
    if (!Config::instance().load("conf/concurrentcache.conf")) {
        LOG_ERROR(NETWORK, "Failed to load config file");
        return 1;
    }

    // ===== 3. 注册配置观察者 =====
    // Logger 观察 log_level 变化，实现热加载
    Config::instance().addObserver("log_level", &Logger::instance());

    // ===== 4. 设置信号处理 =====
    SignalHandler::setup();

    // ===== 5. 创建事件循环 =====
    EventLoop loop;

    // ===== 6. 创建监听套接字 =====
    Socket server;
    int port = Config::instance().getInt("port", 6379);
    if (!server.bind_and_listen(port)) {
        LOG_ERROR(NETWORK, "Failed to start server on port %d", port);
        return 1;
    }

    // ===== 7. 为服务器套接字创建 Channel =====
    Channel server_channel(&loop, server.fd());

    // ===== 8. 设置 accept 回调 =====
    server_channel.set_read_callback([&loop, &server, &server_channel]() {
        // 接受客户端连接
        int client_fd = server.accept();
        if (client_fd >= 0) {
            // 创建新的 Connection
            auto* conn = new Connection(client_fd, &loop);

            // 设置命令处理回调
            conn->set_command_callback([](const RespValue& cmd, Connection* conn) {
                // 提取命令参数
                const auto& arr = cmd.as_array();
                if (arr.empty()) {
                    return;
                }

                std::vector<std::string> args;
                for (size_t i = 0; i < arr.size(); ++i) {
                    args.push_back(arr[i].as_string());
                }

                // 命令名转小写
                if (!args.empty()) {
                    std::string& cmd_name = args[0];
                    std::transform(cmd_name.begin(), cmd_name.end(), cmd_name.begin(),
                                   [](unsigned char c) { return std::tolower(c); });
                }

                // 创建并执行命令
                auto command = CommandFactory::instance().create(args[0]);
                if (command) {
                    std::string response = command->execute(args);
                    conn->send_response(response);
                } else {
                    conn->send_response(RespEncoder::encode_error("ERR unknown command"));
                }
            });
        }
    });

    // ===== 9. 启用读监听 =====
    server_channel.enable_reading();

    LOG_INFO(NETWORK, "Server ready, waiting for connections...");

    // ===== 10. 启动事件循环 =====
    loop.loop();

    LOG_INFO(NETWORK, "=== ConcurrentCache Server Shutdown ===");
    return 0;
}
```
